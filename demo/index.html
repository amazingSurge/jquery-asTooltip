Ajax


$(".tooltip-ajax").asTooltip({
    ajax: function(api) {
        $.ajax({
            type: 'GET',
            url: 'data/data.json',
            dataType: 'json',
            success: function(data, type) {
                api.options.content = data.content;
                api.setContent();
            }
        });
    }
});
